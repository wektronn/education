git add . - добавление в гит
git commit -a  - коммит всех изменений одновременно
git log  - просмотр логов, всех коммитов
git log --oneline - нормальный лог
git log -p - детальный просмотр логов, с изменениями по коммитам
git log -p -2 - детальный просмотр логов, последних 2х коммитов
git checkout - откат на нужный коммит
git checkout master - возврат на последний коммит
git checkout -b название - добавление новой ветки гита
git remote add origin https://github.com/wektronn/education.git - подключение к репозиторию на сайте гитхаба
git push rep-education master - выгрузка на гитхаб проекта
git clone https://github.com/wektronn/education.git
